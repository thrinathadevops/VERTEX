// ── HOW TO USE lib/metadata.ts IN EACH PAGE ─────────────────────
// Add this export to every app/*/page.tsx (server component only)

// Example 1 — Static page (app/blog/page.tsx)
import { PAGE_META } from "@/lib/metadata";
export const metadata = PAGE_META.blog;

// Example 2 — Dynamic [slug] page (app/blog/[slug]/page.tsx)
import { buildMetadata } from "@/lib/metadata";
export async function generateMetadata({ params }: { params: { slug: string } }) {
  const post = await getContentBySlug(params.slug); // your API call
  return buildMetadata({
    title:       post.title,
    description: post.body.replace(/<[^>]*>/g, "").slice(0, 160),
    path:        `/blog/${params.slug}`,
  });
}
